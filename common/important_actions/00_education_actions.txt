action_send_child_to_university_with_vassal = {
	priority = 355
	combine_into_one = yes
	check_create_action = {
		every_child = {
			if = {
				limit = {
					is_adult = no
					num_of_relation_guardian = 0
					age >= childhood_education_start_age
					save_temporary_scope_as = child_student
					root = {
						any_realm_province = {
							has_university_building_trigger = yes
							barony = {
								holder = {
									NOT = {
										this = root
									}
									save_temporary_scope_as = vassal
								}
							}
						}
					}
				}
				try_create_important_action = {
					important_action_type = action_send_child_to_university_with_vassal
					actor = root
					secondary_actor = scope:child_student
					recipient = scope:vassal
				}
			}
		}		
	}
			
	effect = {
		open_interaction_window = {
			interaction = offer_ward_interaction
			actor = root
			secondary_actor = scope:secondary_actor
			recipient = scope:recipient
		}
	}
}
