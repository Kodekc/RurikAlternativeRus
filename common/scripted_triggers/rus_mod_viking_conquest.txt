neighboring_county_or_viking_conquest_trigger = {
	#CHANGE: Vikings can now use Conquest "over sea" only up to 500 distance (Denmark to Norfolk) - was unlimited in vanilla
	OR = {
		trigger_if = {
			limit = {
				$CHARACTER$ = {
					can_use_viking_invasion_cbs_trigger = yes
				}
			}
			is_coastal_county = yes
			#within 500 map distance
			$CHARACTER$ = {
				realm_to_title_distance_squared = { title = prev value < 1500000 }
			}
		}
		#CHANGE: everybody else can use Conquest only on neighbouring and close-over-sea counties (including players, who could apparently target all counties?)
		trigger_else = {
			any_title_to_title_neighboring_and_across_water_county = {
				holder = {
					target_is_same_character_or_above = $CHARACTER$
				}
			}
		}		
	}
}
